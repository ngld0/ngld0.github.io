Breadth First Search #

BFS in Tree #

Recursive #

Iterative (Time O(V + E), Space O(v)) # 

void bfs(TreeNode* root){
    if(root == nullptr) return;
    queue<TreeNode*> que;
    que.push(root);
    while(!que.empty()){
        TreeNode* node = que.front();
        cout << node->val();
        que.pop();
        if(node->left != nullptr) que.push(node->left);
        if(node->right != nullptr) que.push(node->right);
    }
}
