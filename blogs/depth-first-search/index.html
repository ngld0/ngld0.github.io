Depth First Search (DFS) #

Definition #

  • DFS used for traversing tree or graph.
  • The backtracking is used for traversal.
  • Travel from deepest node backtracks it parents.

kinds of DFS in tree #

Recursive (Time O(n), space O(n)) #

  • Preorder (root, left, right) 
    void preorder(struct Node* node){
    if(node == nullptr) 
        return;
    cout<<node->val();
    preorder(node->left);
    preorder(node->right);
}
  • Inorder (left, root, right) 
    void preorder(struct Node* node){
    if(node == nullptr) 
        return;
    preorder(node->left);
    cout<<node->val();
    preorder(node->right);
}
  • Postorder (left,right, root) 
    void preorder(struct Node* node){
    if(node == nullptr) 
        return;
    preorder(node->left);
    preorder(node->right);
    cout<<node->val();
}

Iterative #

  • preorder 
    void iterativePreorder(TreeNode root){
    if(root == nullptr) return;

    stack<TreeNode> st;
    TreeNode cur = root;
    st.push(root);

    while(!st.empty()){

        cur = st.pop();
        cout<< cur->val();

        if(cur->right != nullptr){
            st.push(cur->right);
        }
        if(cur->left != nullptr){
            st.push(cur->left);
        }
    }
}
  • inorder 
    void iterativeInorder(TreeNode root){
    if(root == nullptr) return;
    stack<TreeNode> st;
    TreeNode cur = root;
    while(!st.empty() || cur != nullptr ){
        if(cur != nullptr){
            st.push(cur);
            cur = cur->left;
        }
        else {
            cur = st.pop();
            cout<< cur->val();
            cur = cur->right;
        }
    }
}
  • postorder 
    void iterativePostorder(TreeNode root){
    if(root == nullptr) return;
    stack<TreeNode> st;
    stack<TreeNode> right_st;
    TreeNode cur = root;
    while(!st.empty()){
        if(cur != nullptr){
            if(cur->right != nullptr){
                right_st.push(cur->right);
            }
            st.push(cur);
            cur = cur->left;
        }
        else {
            cur = st.top();
            if(!right_st.empty() && cur->right == right_st.top){
                cur = right_st.pop();
            }
            else {
                cout<< cur->val();
                st.pop();
                cur = nullptr;
            }
        }
    }
}

DFS in Graph (Time O(V + E), Space O (V); E = number of edges, V = number of Vertices) #

  • recursive

    DFS(Graph, Vertex){
    Vertex.visited = true;
    for(child : Graph){
        if(child.visited == false){
            DFS(Graph, child);
        }
    }

}

  • Iterative (non-recursive) DFS is similar with non-recursive BFS, but it has the differents, which are:

    • used stack instead of queue
    • mark discovered after poping the vertex, not before pushing
    • used a reserve iterator instead of an iterator to produce the same result as recursive DFS.
    void iterativeDFS(Graph &graph, int v, vector<bool> &discovered){
    stack<int> st;
    st.push(v);
    while(!st.empty()){
        v=st.top();
        st.pop();
        if(discovered(v)) continue;
        discovered(v) = true;
        cout<< v <<endl;
        // reverse iterator
        for(auto it=graph.adjList[v].rbegin(); it < graph.adjList[v].rbend(); it++){

            int u = *it;
            if(!discovered[u]) st.push(u);
        }
    }
}
